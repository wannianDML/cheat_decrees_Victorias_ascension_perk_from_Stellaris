decision_cheat_modifier_calc = {
	# INFO 手动计算白嫖modifier的“BUG”
	# 因为是手动点击计算的，所以对地区法令的变动不敏感，因此理论上可以颁布完法令然后计算，再取消法令撤回权威力。
	texture = "gfx/interface/icons/decree/decree_cheat_technological_ascendancy.dds"
	possible = {
		is_player = yes
	}
	when_taken = {
		set_variable = {
			name = modifier_cheat_technological_ascendancy_count
			value = 0
		}
		every_scope_state = {
			limit = {
				has_decree = decree_cheat_technological_ascendancy
			}
			prev = {
				change_variable = {
					name = modifier_cheat_technological_ascendancy_count
					add = 1
				}
				# BUG  var 计算结果不是实时更新的
				# 是在下一次点击这个decision的时候才会应用到下面的multiplier
				# 因此玩家需要点击至少两次才能看到效果
				# 没测试是p社bug还是我自己的问题
			}
		}
		if = {
			limit = {
				has_modifier = modifier_cheat_technological_ascendancy
			}
			remove_modifier = modifier_cheat_technological_ascendancy
		}
		add_modifier = {
			name = modifier_cheat_technological_ascendancy
			multiplier = var:modifier_cheat_technological_ascendancy_count
		}
		set_variable = {
			name = modifier_cheat_master_builders_count
			value = 0
		}
		every_scope_state = {
			limit = {
				has_decree = decree_cheat_master_builders
			}
			prev = {
				change_variable = {
					name = modifier_cheat_master_builders_count
					add = 1
				}
			}
		}
		if = {
			limit = {
				has_modifier = modifier_cheat_master_builders
			}
			remove_modifier = modifier_cheat_master_builders
		}
		add_modifier = {
			name = modifier_cheat_master_builders
			multiplier = var:modifier_cheat_master_builders_count
		}
		set_variable = {
			name = modifier_cheat_shared_destiny_count
			value = 0
		}
		every_scope_state = {
			limit = {
				has_decree = decree_cheat_shared_destiny
			}
			prev = {
				change_variable = {
					name = modifier_cheat_shared_destiny_count
					add = 1
				}
			}
		}
		if = {
			limit = {
				has_modifier = modifier_cheat_shared_destiny
			}
			remove_modifier = modifier_cheat_shared_destiny
		}
		add_modifier = {
			name = modifier_cheat_shared_destiny
			multiplier = var:modifier_cheat_shared_destiny_count
		}
		set_variable = {
			name = modifier_cheat_universal_transactions_count
			value = 0
		}
		every_scope_state = {
			limit = {
				has_decree = decree_cheat_universal_transactions
			}
			prev = {
				change_variable = {
					name = modifier_cheat_universal_transactions_count
					add = 1
				}
			}
		}
		if = {
			limit = {
				has_modifier = modifier_cheat_universal_transactions
			}
			remove_modifier = modifier_cheat_universal_transactions
		}
		add_modifier = {
			name = modifier_cheat_universal_transactions
			multiplier = var:modifier_cheat_universal_transactions_count
		}
	}
}
